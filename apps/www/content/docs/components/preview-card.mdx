## Usage

```tsx
import {
  PreviewCard,
  PreviewCardTrigger,
  PreviewCardContent,
} from "@/components/ui/preview-card"
```

```tsx
<PreviewCard>
  <PreviewCardTrigger href="https://example.com">
    Hover me
  </PreviewCardTrigger>
  <PreviewCardContent>
    <p>Preview content here.</p>
  </PreviewCardContent>
</PreviewCard>
```
