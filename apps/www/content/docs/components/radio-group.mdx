## Usage

```tsx
import { RadioGroup, RadioGroupItem } from "@/components/ui/radio-group"
```

```tsx
<RadioGroup defaultValue="default">
  <RadioGroupItem value="default" />
  <RadioGroupItem value="comfortable" />
  <RadioGroupItem value="compact" />
</RadioGroup>
```
