## Usage

```tsx
import {
  Frame,
  FrameDescription,
  FrameHeader,
  FramePanel,
  FrameTitle,
} from "@/components/ui/frame"
```

```tsx
<Frame>
  <FrameHeader>
    <FrameTitle>Project overview</FrameTitle>
    <FrameDescription>Grouped context for related content.</FrameDescription>
  </FrameHeader>
  <FramePanel>Panel content</FramePanel>
</Frame>
```
