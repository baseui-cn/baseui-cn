## Usage

```tsx

  Progress,
  ProgressTrack,
  ProgressIndicator,
  ProgressLabel,
  ProgressValue,
} from "@/components/ui/progress"
```

```tsx
<Progress value={65}>
  <ProgressLabel>Uploading</ProgressLabel>
  <ProgressValue />
</Progress>
```
