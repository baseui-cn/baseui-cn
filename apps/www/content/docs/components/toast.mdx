## Usage

```tsx
import { useToast } from "@/components/ui/toast"
```

```tsx
function MyComponent() {
  const { add } = useToast()

  return (
    <button
      onClick={() =>
        add({
          title: "Saved",
          description: "Your changes have been saved.",
          timeout: 3000,
        })
      }
    >
      Save changes
    </button>
  )
}
```

## Examples

### Variants

Use the `type` property to show toasts with icons and semantic border colors. Supports `success`, `error`, `warning`, and `info`.

<ComponentPreview name="toast-variants" />

### With action (undo)

Add an action button using `actionProps`. The undo pattern closes the original toast and shows a confirmation.

<ComponentPreview name="toast-with-action" />

### Promise

Use `toastManager.promise()` to show a loading spinner, then success or error states. The `loading` type displays a `Loader2` icon.

<ComponentPreview name="toast-promise" />

### Custom data

Pass custom data to toasts via the `data` property.

<ComponentPreview name="toast-custom-data" />

### Toast from drawer

Open a drawer, perform an action that triggers a toast. Closing the toast does **not** close the drawer — they use independent overlay systems.

<ComponentPreview name="toast-from-drawer" />

### Positions

Control where toasts appear on screen using the `position` prop on `ToastProvider`. Supports six positions: `top-left`, `top-center`, `top-right`, `bottom-left`, `bottom-center`, and `bottom-right`.

<ComponentPreview name="toast-positions" />

```tsx
<ToastProvider position="top-center">
  {children}
</ToastProvider>
```

## API

### ToastProvider

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `position` | `"top-left" \| "top-center" \| "top-right" \| "bottom-left" \| "bottom-center" \| "bottom-right"` | `"bottom-right"` | Where toasts appear on screen |
| `limit` | `number` | `5` | Max visible toasts. Excess toasts are stacked behind and animate out with `data-limited` |
| `toastManager` | `ToastManager` | — | Optional external toast manager from `createToastManager()` |

### useToast

Returns the toast manager instance from `Toast.useToastManager()`.

```tsx
const { add, promise, close, toasts } = useToast()
```

### add(options)

| Prop | Type | Description |
|------|------|-------------|
| `title` | `string` | Toast heading |
| `description` | `string` | Supporting text |
| `type` | `"success" \| "error" \| "warning" \| "info" \| "loading"` | Icon and color variant |
| `timeout` | `number` | Auto-dismiss timeout in ms |
| `actionProps` | `object` | Props for the action button (e.g. `{ children: "Undo", onClick: fn }`) |
| `data` | `object` | Custom data attached to the toast |

### promise(promise, options)

| Prop | Type | Description |
|------|------|-------------|
| `loading` | `string \| object` | Message or `{ title, description, type }` shown while loading |
| `success` | `string \| (data) => string \| object` | Message on success |
| `error` | `string \| (err) => string \| object` | Message on error |
