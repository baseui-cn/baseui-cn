## Usage

```tsx
import { Fieldset, FieldsetLegend } from "@/components/ui/fieldset"
```

```tsx
<Fieldset className="space-y-4">
  <FieldsetLegend>Billing details</FieldsetLegend>
  {/* related fields */}
</Fieldset>
```
