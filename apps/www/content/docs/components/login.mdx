## Installation

<InstallTabs
  slug="login"
  addCmd="baseui-cn add login"
  manualDeps="npm install zod"
/>

`baseui-cn add login` installs `components/ui/auth.tsx` and exports `LoginBlock`. Because the block uses success toasts, make sure your app is wrapped once in `ToastProvider`, as shown in the toast docs.

## Usage

```tsx
import { LoginBlock } from "@/components/ui/auth"

<LoginBlock
  onGoogleLogin={() => {}}
  onGithubLogin={() => {}}
  onSubmit={async ({ email, password }) => {
    const isValidDemo = email === "demo@baseui-cn.com" && password === "BaseUI123!"

    if (!isValidDemo) {
      return {
        status: "error",
        message: "Invalid credentials",
        fieldErrors: {
          email: "Try demo@baseui-cn.com",
          password: "Try BaseUI123!",
        },
      }
    }

    return {
      status: "success",
      message: "Logged in successfully",
    }
  }}
/>
```

## Props

<PropsTable>
  <PropRow name="onSubmit" type="(values: LoginValues) => Promise<AuthSubmitResult<LoginFieldName> | void> | AuthSubmitResult<LoginFieldName> | void">
    Optional submit handler. Without it, the block runs a built-in async demo success flow.
  </PropRow>
  <PropRow name="variant" type="'default' | 'bordered' | 'elevated' | 'minimal'" defaultValue="'default'">
    Controls the card presentation.
  </PropRow>
  <PropRow name="size" type="'sm' | 'md' | 'lg'" defaultValue="'md'">
    Controls spacing, typography, and form control sizing.
  </PropRow>
  <PropRow name="animation" type="'none' | 'fade' | 'slide-up' | 'scale'" defaultValue="'fade'">
    Controls entry motion for the block.
  </PropRow>
  <PropRow name="showSocial" type="boolean" defaultValue="true">
    Shows built-in social buttons when provider callbacks are supplied.
  </PropRow>
  <PropRow name="signUpHref" type="string" defaultValue="'/sign-up'">
    Link used in the footer prompt below the login card.
  </PropRow>
  <PropRow name="forgotPasswordHref" type="string" defaultValue="'/forgot-password'">
    Optional recovery link shown beside the password label.
  </PropRow>
  <PropRow name="successToast" type="{ title?: string; description?: string; timeout?: number }">
    Overrides the success toast content used after a successful submit.
  </PropRow>
</PropsTable>
