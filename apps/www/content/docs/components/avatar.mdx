## Usage

```tsx
import {
  Avatar,
  AvatarImage,
  AvatarFallback,
  AvatarBadge,
  AvatarGroup,
  AvatarGroupCount,
} from "@/components/ui/avatar"
```

```tsx
<Avatar>
  <AvatarImage src="/photo.jpg" alt="Aria Chen" />
  <AvatarFallback>AC</AvatarFallback>
</Avatar>
```
